175 float GraficadorDesempenio::newton(int n, float *x, float *y, float xi)
176 {
177  float b[n][n];
178  float xt;
179  float yi;
180 
181  //inicializo tabla de diferencias
182  for (int i=0; i<n; i++)
183  b[i][0]=y[i];
184 
185  //calculo la tabla de diferencias
186  for(int j=1; j<n; j++)
187  for(int i=0; i<n-j; i++)
188  b[i][j]=(b[i+1][j-1]-b[i][j-1])/(x[i+j]-x[i]);
189 
190 
191  //interpolo con el polinomio
192  xt=1;
193  yi=b[0][0];
194  for(int j=0; j<n-1; j++){
195  xt=xt*(xi-x[j]);
196  yi=yi+b[0][j+1]*xt;
197  }
198
