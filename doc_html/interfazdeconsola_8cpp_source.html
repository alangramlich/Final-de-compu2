377 void InterfazDeConsola::mostrarEcuacion(int n, float *x, float *y, float xi)
378 {
379  float b[n][n];
380  float xt;
381  float yi;
382  string polinomios, ecuacion;
383  string aux;
384  ecuacion="y=";
385  //inicializo tabla de diferencias
386  for (int i=0; i<n; i++)
387  b[i][0]=y[i];
388 
389  //calculo la tabla de diferencias
390  for(int j=1; j<n; j++)
391  for(int i=0; i<n-j; i++)
392  b[i][j]=(b[i+1][j-1]-b[i][j-1])/(x[i+j]-x[i]);
393  //interpolo con el polinomio
394  xt=1;
395  yi=b[0][0];
396  ecuacion+=to_string((int)yi);
397  for(int j=0; j<n-1; j++){
398  xt=xt*(xi-x[j]);
399  if (x[j]<0)
400  polinomios+="(x"+to_string((int)x[j])+")";
401  else
402  polinomios+="(x-"+to_string((int)x[j])+")";
403  yi=yi+b[0][j+1]*xt;
404  aux=to_string(b[0][j+1]);
405  bool salir=false;
406  while (!salir){
407  if(aux[aux.size()-1]=='0' or aux[aux.size()-1]=='.')
408  aux.pop_back();
409  else
410  salir=true;
411  }
412  if (b[0][j+1]==0)
413  ecuacion+="+(0)*";
414  else{
415  if (b[0][j+1]>=0)
416  ecuacion+="+("+aux+')'+"*"+polinomios;
417  else
418  ecuacion+='('+aux+')'+"*"+polinomios;
419  }
420  }

422 }
