234 void JuegoDeLaMemoria::mousePressEvent(QMouseEvent *event)
235 {
236  if(event->button() == Qt::LeftButton && cartasSeleccionadas[1] == nullptr)

238  if (primerCarta==true){
239  tiempoDeInicio=time(0);
240  primerCarta=false;

242  double posX = event->x() * anchoMesaConCartas / width() - margen;
243  double posY = (height() - event->y()) * altoMesaConCartas / height() - margen;
244  cx = coordenadaAPosicionCuantizada(posX, anchoMesaConCartas);
245  cy = coordenadaAPosicionCuantizada(posY, altoMesaConCartas);
246  if (CASO_TRES==false){
247  if (cx>=0 && cy>=0

249  && !mesaConCartas(unsigned(cy),unsigned(cx))->getVisible()) {
250  if (cartasSeleccionadas[0] == nullptr) {
251  cartasSeleccionadas[0] = mesaConCartas(unsigned(cy),unsigned(cx));
252  } else {
253  cartasSeleccionadas[1] = mesaConCartas(unsigned(cy),unsigned(cx));

256  // Marcamos que las encontramos
